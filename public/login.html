async function login() { const user = document.getElementById("user").value; const password = document.getElementById("password").value; const res = await fetch("/login", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ user, password }) }); const data = await res.json(); const msg = document.getElementById("msg"); if (data.success) { msg.innerHTML = "Login aprovado ✅"; msg.style.color = "green"; } else { msg.innerHTML = "Login inválido ❌"; msg.style.color = "red"; } }
